float4 TemporalUpscale(float2 uv) {

    float2 jitteredUV = uv + g_Jitter / g_LowResolution;
    float4 current = g_LowRes.Sample(sampler, jitteredUV);
    
    // Reproject history

    float4 minColor, maxColor;

    history = clamp(history, minColor, maxColor);

    return lerp(history, current, blend);
}

## 1052. Lanczos Reconstruction
**Novelty Class:** Significant Improvement

High-quality spatial upscaling.

```hlsl
float Lanczos(float x, float a) {
    if (x == 0) return 1;
    if (abs(x) >= a) return 0;
    float pi_x = PI * x;
    return (a * sin(pi_x) * sin(pi_x / a)) / (pi_x * pi_x);
}

float4 LanczosUpscale(float2 uv) {
    float2 texelSize = 1.0 / g_SourceResolution;
    float2 center = uv * g_SourceResolution;
    int2 base = int2(center) - 2;
    
    float4 result = 0;
    float weightSum = 0;
    
    [unroll] for (int y = 0; y < 5; y++) {
        [unroll] for (int x = 0; x < 5; x++) {
            float2 samplePos = float2(base + int2(x, y)) + 0.5;
            float2 delta = center - samplePos;
            float weight = Lanczos(delta.x, 3) * Lanczos(delta.y, 3);
            
            result += g_Source.SampleLevel(sampler, samplePos * texelSize, 0) * weight;
            weightSum += weight;
        }
    }
    
    return result / weightSum;
}
```

## 1053. Edge-Directed Interpolation
**Novelty Class:** Significant Improvement

```hlsl
float4 EdgeDirectedUpscale(float2 uv) {
    // Detect edge direction via gradient
    float2 grad = ComputeGradient(uv);
    float2 edgeDir = normalize(float2(-grad.y, grad.x));
    
    // Sample along edge
    float2 texel = 1.0 / g_Resolution;
    float4 a = g_Input.Sample(sampler, uv + edgeDir * texel);
    float4 b = g_Input.Sample(sampler, uv - edgeDir * texel);
    
    // Blend based on edge strength
    float edgeStrength = length(grad);
    float4 edgeSample = lerp(a, b, 0.5);
    float4 bilinear = g_Input.Sample(sampler, uv);
    
    return lerp(bilinear, edgeSample, saturate(edgeStrength * 10));
}
```

## 1055. FSR-Style Spatial Upscaling
**Novelty Class:** Significant Improvement

EASU + RCAS two-pass upscaling.

```hlsl
// Edge Adaptive Spatial Upsampling
float4 EASU(float2 uv) {
    float2 pos = uv * g_InputSize;
    int2 base = int2(pos);
    float2 frac = frac(pos);
    
    // Gather 12 samples
    float4 samples[12];
    GatherSamples(base, samples);
    
    // Detect edges
    float2 dir = ComputeEdgeDirection(samples);
    float len = length(dir);
    
    // Rotate kernel along edge
    float4 result = 0;
    [unroll] for (int i = 0; i < 12; i++) {
        float weight = ComputeRotatedWeight(i, frac, dir, len);
        result += samples[i] * weight;
    }
    
    return result;
}

// Robust Contrast Adaptive Sharpening
float4 RCAS(float2 uv) {
    float4 center = g_Input.Sample(sampler, uv);
    float4 north = g_Input.Sample(sampler, uv + float2(0, -1) / g_Resolution);
    float4 south = g_Input.Sample(sampler, uv + float2(0,  1) / g_Resolution);
    float4 east  = g_Input.Sample(sampler, uv + float2( 1, 0) / g_Resolution);
    float4 west  = g_Input.Sample(sampler, uv + float2(-1, 0) / g_Resolution);
    
    float4 minNS = min(north, south);
    float4 maxNS = max(north, south);
    float4 minEW = min(east, west);
    float4 maxEW = max(east, west);
    
    float4 peak = min(minNS, minEW) / max(maxNS, maxEW);
    float4 w = sqrt(saturate(peak)) * g_Sharpness;
    
    return (center + (north + south + east + west) * w) / (1 + 4 * w);
}
```
