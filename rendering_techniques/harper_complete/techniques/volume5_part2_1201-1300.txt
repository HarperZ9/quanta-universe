                float3 cellPos = hash3(i + neighbor);
                float3 diff = neighbor + cellPos - f;
                float dist = length(diff);
                
                if (dist < result.distance1) {
                    result.distance2 = result.distance1;
                    result.distance1 = dist;
                    result.cellId = i + neighbor;
                } else if (dist < result.distance2) {
                    result.distance2 = dist;
                }
            }
        }
    }
    
    return result;
}
```

float3 ProceduralSky(float3 viewDir, float3 sunDir) {
    // Rayleigh scattering
    float cosTheta = dot(viewDir, sunDir);
    float rayleighPhase = 0.75 * (1.0 + cosTheta * cosTheta);
    
    // Mie scattering
    float g = 0.76;
    float miePhase = (1.0 - g*g) / pow(1.0 + g*g - 2.0*g*cosTheta, 1.5);
    
    // Optical depth
    float zenith = acos(saturate(viewDir.y));
    float rayleighDepth = exp(-zenith * 0.15);
    float mieDepth = exp(-zenith * 0.1);
    
    // Scattering colors
    float3 rayleighColor = float3(0.2, 0.5, 1.0);
    float3 mieColor = float3(1.0, 0.95, 0.8);
    
    return rayleighColor * rayleighPhase * rayleighDepth +
           mieColor * miePhase * mieDepth;
}

## 1302. Bandwidth Reduction

// Pack multiple attributes into fewer channels
float4 PackGBuffer(float3 albedo, float3 normal, float roughness, float metallic) {
    // RGB: Albedo (8-bit each)
    // A: Packed normal XY (4-bit each) + roughness (4-bit) + metallic (4-bit)
