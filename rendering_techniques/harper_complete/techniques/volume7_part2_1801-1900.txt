    
    g_ClothPrevPositions[tid] = pos;
    g_ClothPositions[tid] = newPos;
}

[numthreads(256, 1, 1)]
void CS_ClothConstraints(uint tid : SV_DispatchThreadID) {
    ClothConstraint c = g_Constraints[tid];
    
    float3 p0 = g_ClothPositions[c.idx0];
    float3 p1 = g_ClothPositions[c.idx1];
    
    float3 delta = p1 - p0;
    float dist = length(delta);
    float error = dist - c.restLength;
    
    float3 correction = (delta / dist) * error * 0.5;
    
    float w0 = g_ClothInvMass[c.idx0];
    float w1 = g_ClothInvMass[c.idx1];
    float totalW = w0 + w1;
    
    if (w0 > 0) InterlockedAdd(g_ClothPositions[c.idx0], correction * (w0 / totalW));
    if (w1 > 0) InterlockedAdd(g_ClothPositions[c.idx1], -correction * (w1 / totalW));
}

```hlsl
void CS_HairSimulation(uint strandId : SV_GroupID, uint vertId : SV_GroupThreadID) {

    // Distance constraint to parent
    float3 toParent = parentPos - newPos;
    float dist = length(toParent);
    if (dist > g_SegmentLength) {
        newPos = parentPos - normalize(toParent) * g_SegmentLength;
    }
    
    g_HairPrevPositions[globalId] = pos;
    g_HairPositions[globalId] = newPos;
}
```

## 1803. Fluid Surface Rendering
**Novelty Class:** Significant Improvement

Screen-space fluid rendering for SPH.

```hlsl
// Depth pass: splat particle spheres
float PS_FluidDepth(float4 pos : SV_Position, float2 uv : TEXCOORD) : SV_Depth {
    float dist = length(uv - 0.5) * 2;
    if (dist > 1) discard;
    
    float sphereDepth = sqrt(1 - dist * dist);
    return pos.z - sphereDepth * g_ParticleRadius / g_FarPlane;
}

// Smooth depth with bilateral filter
float PS_SmoothFluidDepth(float2 uv : TEXCOORD) : SV_Target {
    float centerDepth = g_FluidDepth.Sample(sampler, uv);
    if (centerDepth >= 1) return centerDepth;
    
    float sum = 0;
    float wsum = 0;
    
    for (int y = -5; y <= 5; y++) {
        for (int x = -5; x <= 5; x++) {
            float2 offset = float2(x, y) / g_Resolution;
            float sampleDepth = g_FluidDepth.Sample(sampler, uv + offset);
            
            float rangeWeight = exp(-abs(sampleDepth - centerDepth) * g_DepthFalloff);
            float spatialWeight = exp(-(x*x + y*y) * g_SpatialFalloff);
            float w = rangeWeight * spatialWeight;
            
            sum += sampleDepth * w;
            wsum += w;
        }
    }
    
    return sum / wsum;
}

// Final shading
float4 PS_FluidShading(float2 uv : TEXCOORD) : SV_Target {
    float depth = g_SmoothedDepth.Sample(sampler, uv);
    if (depth >= 1) discard;
    
    // Reconstruct normal from depth
    float3 normal = ReconstructNormalFromDepth(uv, depth);
    
    float3 worldPos = ReconstructWorldPos(uv, depth);
    float3 viewDir = normalize(g_CameraPos - worldPos);
    
    // Fresnel + refraction
    float fresnel = pow(1 - saturate(dot(normal, viewDir)), 5);
    float3 reflection = g_EnvMap.Sample(sampler, reflect(-viewDir, normal)).rgb;
    
    float3 refractDir = refract(-viewDir, normal, 1.0 / 1.33);
    float2 refractUV = uv + refractDir.xy * g_RefractionStrength;
    float3 refraction = g_SceneColor.Sample(sampler, refractUV).rgb * g_WaterColor;
    
    return float4(lerp(refraction, reflection, fresnel), 1);
}
```

## 1851. Screen-Space Reflections Advanced
**Novelty Class:** Significant Improvement

Hi-Z accelerated SSR with roughness.

```hlsl
float4 SSR(float2 uv, float3 worldPos, float3 normal, float roughness) {
    float3 viewDir = normalize(worldPos - g_CameraPos);
    float3 reflectDir = reflect(viewDir, normal);
    
    // Jitter for rough surfaces
    if (roughness > 0.05) {
        float2 noise = g_BlueNoise.Sample(sampler, uv * 16).xy;
        float3 tangent = normalize(cross(normal, float3(0, 1, 0)));
        float3 bitangent = cross(normal, tangent);
        
        float2 diskSample = ConcentricDiskSample(noise) * roughness;
        reflectDir = normalize(reflectDir + tangent * diskSample.x + bitangent * diskSample.y);
    }
    
    // Hi-Z ray march
    float3 rayPos = worldPos + reflectDir * 0.1;
    float3 rayDir = reflectDir;
    
    for (int mip = 0; mip < 8; mip++) {
        float2 rayUV = WorldToScreen(rayPos).xy;
        float sceneDepth = g_HiZ.SampleLevel(sampler, rayUV, mip);
        float rayDepth = WorldToDepth(rayPos);
        
        if (rayDepth > sceneDepth) {
            if (mip == 0) {
                // Hit found
                return g_SceneColor.Sample(sampler, rayUV);
            }
            // Refine at lower mip
            rayPos -= rayDir * GetStepSize(mip);
            mip -= 2;
        } else {
            rayPos += rayDir * GetStepSize(mip);
        }
    }
    
    return float4(0, 0, 0, 0); // Miss - fallback to probe
}
```

```hlsl
float4 PS_Refraction(float2 uv : TEXCOORD, float3 worldPos : WORLDPOS, 
                     float3 normal : NORMAL) : SV_Target {
```

## 1853. Iridescence
**Novelty Class:** Significant Improvement

Thin-film interference for oil slicks and soap bubbles.

```hlsl
float3 ThinFilmIridescence(float cosTheta, float filmThickness) {
    // Optical path difference
    float n = 1.33; // Film refractive index
    float d = filmThickness; // nm
    
    float3 wavelengths = float3(650, 510, 475); // RGB wavelengths in nm
    
    float3 phase;
    for (int i = 0; i < 3; i++) {
        float opd = 2 * n * d * sqrt(1 - sin(acos(cosTheta)) * sin(acos(cosTheta)) / (n * n));
        phase[i] = 2 * PI * opd / wavelengths[i];
    }
    
    // Interference intensity
    float3 interference = 0.5 + 0.5 * cos(phase);
    
    return interference;
}
```

## 1902. Vector Graphics
**Novelty Class:** Significant Improvement

GPU-accelerated path rendering.

```hlsl
float4 PS_VectorShape(float2 uv : TEXCOORD) : SV_Target {
    // Evaluate SDF for shape
    float dist = EvaluatePathSDF(uv);
    
    // Anti-aliased edge
    float edgeWidth = fwidth(dist);
    float alpha = 1 - smoothstep(-edgeWidth, edgeWidth, dist);
    
    // Stroke
    if (g_StrokeWidth > 0) {
        float strokeDist = abs(dist) - g_StrokeWidth;
        float strokeAlpha = 1 - smoothstep(-edgeWidth, edgeWidth, strokeDist);
        
        float3 color = lerp(g_FillColor.rgb, g_StrokeColor.rgb, 
                           strokeAlpha * step(dist, 0));
        alpha = max(alpha, strokeAlpha);
        
        return float4(color, alpha);
    }
    
    return float4(g_FillColor.rgb, alpha * g_FillColor.a);
}

float EvaluatePathSDF(float2 p) {
    float minDist = 1e10;
    
    for (uint i = 0; i < g_SegmentCount; i++) {
        PathSegment seg = g_Segments[i];
        float dist;
        
        switch (seg.type) {
            case SEG_LINE: dist = LineDistance(p, seg.p0, seg.p1); break;
            case SEG_QUADRATIC: dist = QuadraticBezierDistance(p, seg.p0, seg.p1, seg.p2); break;
            case SEG_CUBIC: dist = CubicBezierDistance(p, seg.p0, seg.p1, seg.p2, seg.p3); break;
        }
        
        minDist = min(minDist, dist);
    }
    
    return minDist * (IsInside(p) ? -1 : 1);
}
```

## 1903. Rounded Rectangle
**Novelty Class:** Significant Improvement

Analytical SDF for UI elements.

```hlsl
float RoundedRectSDF(float2 p, float2 size, float4 radii) {
    // Select corner radius based on quadrant
    float2 q = abs(p) - size;
    float r = (p.x > 0) ? ((p.y > 0) ? radii.x : radii.y) 
                        : ((p.y > 0) ? radii.w : radii.z);
    
    q += r;
    return min(max(q.x, q.y), 0) + length(max(q, 0)) - r;
}

float4 PS_RoundedRect(float2 uv : TEXCOORD) : SV_Target {
    float2 p = (uv - 0.5) * g_Size;
    float dist = RoundedRectSDF(p, g_Size * 0.5 - g_BorderWidth, g_CornerRadii);
    
    float edgeWidth = fwidth(dist);
    
    // Fill
    float fillAlpha = 1 - smoothstep(-edgeWidth, edgeWidth, dist);
    
    // Border
    float borderDist = abs(dist + g_BorderWidth * 0.5) - g_BorderWidth * 0.5;
    float borderAlpha = 1 - smoothstep(-edgeWidth, edgeWidth, borderDist);
    
    float3 color = lerp(g_FillColor.rgb, g_BorderColor.rgb, borderAlpha);
    float alpha = max(fillAlpha, borderAlpha);
    
    // Drop shadow
    float shadowDist = RoundedRectSDF(p - g_ShadowOffset, g_Size * 0.5, g_CornerRadii);
    float shadowAlpha = 1 - smoothstep(0, g_ShadowBlur, shadowDist);
    
    return float4(color, alpha) + float4(0, 0, 0, shadowAlpha * g_ShadowOpacity) * (1 - alpha);
}
```
