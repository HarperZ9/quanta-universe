## 801. Meshlet Sphere Culling
**Novelty Class:** Significant Improvement

Tighter culling via bounding sphere per meshlet.

```hlsl
bool CullMeshletSphere(Meshlet m, float4 frustumPlanes[6]) {
    float3 center = mul(g_WorldMatrix, float4(m.boundingSphere.xyz, 1)).xyz;
    float radius = m.boundingSphere.w * g_MaxScale;
    
    [unroll] for (int i = 0; i < 6; i++) {
        if (dot(frustumPlanes[i].xyz, center) + frustumPlanes[i].w < -radius)
            return true; // Culled
    }
    return false;
}
```

**Result:** 15% tighter culling than AABB.

## 803. Hierarchical Meshlet LOD
**Novelty Class:** Patent-Worthy

Tree structure of meshlet clusters for seamless LOD.

```hlsl
struct MeshletCluster {
    uint childCount;
    uint children[8];
    float screenSizeThreshold;
    uint meshletStart, meshletCount;
};

void TraverseLODTree(uint nodeId, float screenSize) {
    MeshletCluster c = g_Clusters[nodeId];
    
    if (screenSize < c.screenSizeThreshold || c.childCount == 0) {
        // Emit this level's meshlets
        for (uint i = 0; i < c.meshletCount; i++)
            EmitMeshlet(c.meshletStart + i);
    } else {
        // Recurse to children
        for (uint i = 0; i < c.childCount; i++)
            TraverseLODTree(c.children[i], screenSize);
    }
}
```

## 805. Meshlet Compression
**Novelty Class:** Significant Improvement

Delta-encoded indices, quantized positions per meshlet.

```hlsl
struct CompressedMeshlet {
    uint8_t localIndices[378];  // 126 triangles, 3 bytes each
    uint16_t vertexDeltas[64*3]; // 16-bit position deltas
    float3 origin;
    float scale;
};

float3 DecompressVertex(CompressedMeshlet m, uint localVert) {
    int3 delta = int3(
        m.vertexDeltas[localVert*3],
        m.vertexDeltas[localVert*3+1],
        m.vertexDeltas[localVert*3+2]
    );
    return m.origin + float3(delta) * m.scale / 65535.0;
}
```

## 858. Ray Differentials
**Novelty Class:** Significant Improvement

Track ray footprint for LOD/filtering.

```hlsl
struct RayDifferential {
    Ray ray;
    float3 dOdx, dOdy; // Origin derivatives
    float3 dDdx, dDdy; // Direction derivatives
};

float ComputeMipLevel(RayDifferential rd, Hit hit) {
    // Project differentials to hit surface
    float3 dPdx = rd.dOdx + hit.t * rd.dDdx;
    float3 dPdy = rd.dOdy + hit.t * rd.dDdy;
    
    // Convert to texture space
    float2 dUVdx = WorldToUV(dPdx, hit);
    float2 dUVdy = WorldToUV(dPdy, hit);
    
    float footprint = max(length(dUVdx), length(dUVdy));
    return log2(footprint * g_TextureSize);
}
```
