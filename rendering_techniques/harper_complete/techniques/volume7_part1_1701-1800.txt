## 1701. Rayleigh-Mie Scattering
**Novelty Class:** Patent-Worthy

Full atmospheric scattering with precomputation.

```hlsl
float3 ComputeAtmosphericScattering(float3 rayDir, float3 sunDir) {
    float mu = dot(rayDir, sunDir);
    
    // Rayleigh phase function
    float phaseR = 3.0 / (16.0 * PI) * (1.0 + mu * mu);
    
    // Mie phase function (Henyey-Greenstein)
    float g = 0.76;
    float phaseM = 3.0 / (8.0 * PI) * ((1.0 - g*g) * (1.0 + mu*mu)) / 
                   ((2.0 + g*g) * pow(1.0 + g*g - 2.0*g*mu, 1.5));
    
    // Sample precomputed transmittance LUT
    float altitude = length(g_CameraPos) - g_EarthRadius;
    float zenith = acos(clamp(rayDir.y, -1, 1));
    float2 lutUV = float2(altitude / g_AtmosphereHeight, zenith / PI);
    
    float3 transmittance = g_TransmittanceLUT.Sample(sampler, lutUV).rgb;
    float3 inscattering = g_InscatteringLUT.Sample(sampler, float3(lutUV, mu * 0.5 + 0.5)).rgb;
    
    return inscattering * (g_RayleighCoeff * phaseR + g_MieCoeff * phaseM);
}
```

[numthreads(8, 8, 1)]
void CS_InjectFogDensity(uint3 tid : SV_DispatchThreadID) {
    float3 froxelCenter = FroxelToWorld(tid);
    
    float density = g_GlobalFogDensity;
    
    // Height-based falloff
    density *= exp(-max(0, froxelCenter.y - g_FogBaseHeight) * g_FogHeightFalloff);
    
    // Noise-based variation
    density *= 0.5 + 0.5 * FBM(froxelCenter * 0.01 + g_Time * 0.1, 4);
    
    // Local fog volumes
    for (uint i = 0; i < g_FogVolumeCount; i++) {
        FogVolume vol = g_FogVolumes[i];
        float3 localPos = mul(vol.worldToLocal, float4(froxelCenter, 1)).xyz;
        if (all(abs(localPos) < 0.5)) {
            density += vol.density * (1 - length(localPos) * 2);
        }
    }
    
    g_FogDensity[tid] = density;
}

[numthreads(8, 8, 1)]
void CS_AccumulateFog(uint3 tid : SV_DispatchThreadID) {
    float3 accumScattering = 0;
    float accumTransmittance = 1;
    
    for (uint z = 0; z < FROXEL_DEPTH; z++) {
        uint3 froxelIdx = uint3(tid.xy, z);
        float density = g_FogDensity[froxelIdx];
        float sliceDepth = FroxelSliceThickness(z);
        
        float extinction = density * sliceDepth;
        float transmittance = exp(-extinction);
        
        float3 lighting = SampleFroxelLighting(froxelIdx);
        float3 scattering = lighting * density * (1 - transmittance) / max(extinction, 0.0001);
        
        accumScattering += scattering * accumTransmittance;
        accumTransmittance *= transmittance;
        
        g_FogAccum[froxelIdx] = float4(accumScattering, accumTransmittance);
    }
}

**Novelty Class:** Significant Improvement

float4 PS_GodRays(float2 uv : TEXCOORD) : SV_Target {
    float2 lightScreenPos = WorldToScreen(g_SunDirection * 10000).xy;
    float2 rayDir = normalize(lightScreenPos - uv);
    
    float3 accumLight = 0;
    float accumDensity = 0;
    
    const int SAMPLES = 64;
    float stepSize = length(lightScreenPos - uv) / SAMPLES;
    
    for (int i = 0; i < SAMPLES; i++) {
        float2 sampleUV = uv + rayDir * stepSize * i;
        
        if (all(sampleUV > 0) && all(sampleUV < 1)) {
            float depth = g_Depth.Sample(sampler, sampleUV);
            float3 worldPos = ReconstructWorldPos(sampleUV, depth);
            
            // Shadow test
            float shadow = SampleShadowMap(worldPos);
            
            // Accumulate lit fog
            float fogDensity = ComputeFogDensity(worldPos);
            accumLight += shadow * fogDensity * g_SunColor;
            accumDensity += fogDensity;
        }
    }
    
    return float4(accumLight / SAMPLES, 1);
}

## 1704. Cloud Raymarching
**Novelty Class:** Patent-Worthy

Stratocumulus clouds with temporal reprojection.

```hlsl
float4 RaymarchClouds(float3 rayOrigin, float3 rayDir) {
    float3 accumColor = 0;
    float accumTransmittance = 1;
    
    float tMin, tMax;
    if (!IntersectCloudLayer(rayOrigin, rayDir, tMin, tMax)) 
        return float4(0, 0, 0, 1);
    
    float t = tMin + g_BlueNoise.Sample(sampler, uv).r * g_StepSize;
    
    for (int i = 0; i < MAX_STEPS && t < tMax && accumTransmittance > 0.01; i++) {
        float3 pos = rayOrigin + rayDir * t;
        
        float density = SampleCloudDensity(pos);
        
        if (density > 0.001) {
            float3 lighting = SampleCloudLighting(pos, g_SunDir);
            
            float extinction = density * g_ExtinctionCoeff;
            float transmittance = exp(-extinction * g_StepSize);
            
            float3 scattering = lighting * density * (1 - transmittance);
            accumColor += scattering * accumTransmittance;
            accumTransmittance *= transmittance;
        }
        
        t += g_StepSize * (density > 0 ? 1 : 2); // Adaptive stepping
    }
    
    return float4(accumColor, accumTransmittance);
}

float SampleCloudDensity(float3 pos) {
    // Weather map lookup
    float2 weatherUV = pos.xz * 0.00001;
    float4 weather = g_WeatherMap.Sample(sampler, weatherUV);
    float coverage = weather.r;
    float cloudType = weather.g;
    
    // Height gradient
    float heightFrac = (pos.y - g_CloudBaseHeight) / g_CloudThickness;
    float heightGradient = CloudHeightGradient(heightFrac, cloudType);
    
    // Base shape
    float baseNoise = g_BaseShapeNoise.Sample(sampler, pos * 0.0003).r;
    float baseDensity = Remap(baseNoise, 1 - coverage, 1, 0, 1) * heightGradient;
    
    // Detail erosion
    if (baseDensity > 0) {
        float detailNoise = g_DetailNoise.Sample(sampler, pos * 0.003).r;
        baseDensity -= detailNoise * 0.3 * (1 - heightFrac);
    }
    
    return max(0, baseDensity);
}
```

**Novelty Class:** Significant Improvement

float4 PS_Underwater(float2 uv : TEXCOORD) : SV_Target {
    float depth = g_Depth.Sample(sampler, uv);
    float3 worldPos = ReconstructWorldPos(uv, depth);
    float waterDepth = g_WaterSurfaceHeight - worldPos.y;
    
    float3 sceneColor = g_SceneColor.Sample(sampler, uv).rgb;
    
    if (g_CameraPos.y < g_WaterSurfaceHeight) {
        // Camera underwater
        float viewDist = length(worldPos - g_CameraPos);
        
        // Exponential fog absorption
        float3 absorption = exp(-viewDist * g_WaterAbsorption);
        
        // Scattering color
        float3 scattering = g_WaterColor * (1 - absorption);
        
        // Caustics on underwater surfaces
        if (waterDepth > 0) {
            float2 causticUV = worldPos.xz * 0.1 + g_Time * 0.05;
            float caustic = g_CausticTexture.Sample(sampler, causticUV).r;
            caustic *= exp(-waterDepth * 0.1);
            sceneColor += caustic * g_SunColor;
        }
        
        sceneColor = sceneColor * absorption + scattering;
    }
    
    return float4(sceneColor, 1);
}
```

## 1751. GPU Particle Simulation
**Novelty Class:** Significant Improvement

Compute-driven particle update.

```hlsl
struct Particle {
    float3 position;
    float3 velocity;
    float age;
    float lifetime;
    float size;
    float4 color;
};

[numthreads(256, 1, 1)]
void CS_UpdateParticles(uint tid : SV_DispatchThreadID) {
    Particle p = g_Particles[tid];
    
    if (p.age >= p.lifetime) {
        // Dead particle - recycle
        if (g_EmitCount > 0) {
            InterlockedAdd(g_EmitCount, -1);
            p = EmitNewParticle(tid);
        }
        return;
    }
    
    // Physics update
    p.velocity += g_Gravity * g_DeltaTime;
    p.velocity *= 1 - g_Drag * g_DeltaTime;
    
    // Collision
    float3 newPos = p.position + p.velocity * g_DeltaTime;
    if (newPos.y < 0) {
        newPos.y = 0;
        p.velocity.y *= -g_Bounce;
    }
    
    p.position = newPos;
    p.age += g_DeltaTime;
    
    // Age-based color/size
    float t = p.age / p.lifetime;
    p.color = lerp(g_StartColor, g_EndColor, t);
    p.size = lerp(g_StartSize, g_EndSize, t);
    
    g_Particles[tid] = p;
}
```

## 1752. Particle Sorting
**Novelty Class:** Significant Improvement

Bitonic sort for depth-ordered particles.

```hlsl
[numthreads(256, 1, 1)]
void CS_BitonicSort(uint tid : SV_DispatchThreadID) {
    uint i = tid;
    
    for (uint k = 2; k <= g_ParticleCount; k *= 2) {
        for (uint j = k / 2; j > 0; j /= 2) {
            uint ixj = i ^ j;
            
            if (ixj > i) {
                float depthI = g_ParticleDepths[i];
                float depthJ = g_ParticleDepths[ixj];
                
                bool ascending = (i & k) == 0;
                bool shouldSwap = ascending ? (depthI > depthJ) : (depthI < depthJ);
                
                if (shouldSwap) {
                    // Swap
                    g_ParticleDepths[i] = depthJ;
                    g_ParticleDepths[ixj] = depthI;
                    
                    Particle temp = g_Particles[i];
                    g_Particles[i] = g_Particles[ixj];
                    g_Particles[ixj] = temp;
                }
            }
            
            GroupMemoryBarrierWithGroupSync();
        }
    }
}
```

## 1753. Billboard Generation
**Novelty Class:** Significant Improvement

Mesh shader particle billboards.

```hlsl
struct ParticleVertex {
    float4 position : SV_Position;
    float2 uv : TEXCOORD0;
    float4 color : COLOR;
};

[outputtopology("triangle")]
[numthreads(32, 1, 1)]
void MS_ParticleBillboard(
    uint tid : SV_GroupThreadID,
    uint gid : SV_GroupID,
    out vertices ParticleVertex verts[128],
    out indices uint3 tris[64])
{
    SetMeshOutputCounts(gid < g_ParticleCount / 32 ? 128 : 0, 
                        gid < g_ParticleCount / 32 ? 64 : 0);
    
    uint particleIdx = gid * 32 + tid;
    if (particleIdx >= g_ParticleCount) return;
    
    Particle p = g_Particles[particleIdx];
    
    float3 right = g_CameraRight * p.size;
    float3 up = g_CameraUp * p.size;
    
    // 4 vertices per particle
    uint baseVert = tid * 4;
    verts[baseVert + 0].position = mul(g_ViewProj, float4(p.position - right - up, 1));
    verts[baseVert + 0].uv = float2(0, 1);
    verts[baseVert + 1].position = mul(g_ViewProj, float4(p.position + right - up, 1));
    verts[baseVert + 1].uv = float2(1, 1);
    verts[baseVert + 2].position = mul(g_ViewProj, float4(p.position + right + up, 1));
    verts[baseVert + 2].uv = float2(1, 0);
    verts[baseVert + 3].position = mul(g_ViewProj, float4(p.position - right + up, 1));
    verts[baseVert + 3].uv = float2(0, 0);
    
    for (int i = 0; i < 4; i++)
        verts[baseVert + i].color = p.color;
    
    // 2 triangles per particle
    uint baseTri = tid * 2;
    tris[baseTri + 0] = uint3(baseVert, baseVert + 1, baseVert + 2);
    tris[baseTri + 1] = uint3(baseVert, baseVert + 2, baseVert + 3);
}

## 1754. Soft Particles
**Novelty Class:** Significant Improvement

Depth-based edge fade.

```hlsl
float4 PS_SoftParticle(float4 pos : SV_Position, float2 uv : TEXCOORD, 
                       float4 color : COLOR) : SV_Target {
    float4 texColor = g_ParticleTexture.Sample(sampler, uv);
    
    // Scene depth
    float sceneDepth = LinearizeDepth(g_Depth.Load(int3(pos.xy, 0)));
    float particleDepth = LinearizeDepth(pos.z);
    
    // Soft edge fade
    float depthDiff = sceneDepth - particleDepth;
    float softness = saturate(depthDiff / g_SoftnessDistance);
    
    return texColor * color * softness;
}
```

## 1801. GPU Cloth Simulation
**Novelty Class:** Patent-Worthy

Position-based dynamics for cloth.

```hlsl
[numthreads(256, 1, 1)]
void CS_ClothSimulation(uint tid : SV_DispatchThreadID) {
    float3 pos = g_ClothPositions[tid];
    float3 prevPos = g_ClothPrevPositions[tid];
    float invMass = g_ClothInvMass[tid];
    
    if (invMass == 0) return; // Pinned vertex
    
    // Verlet integration
    float3 velocity = pos - prevPos;
    float3 newPos = pos + velocity * g_Damping + g_Gravity * g_DeltaTime * g_DeltaTime;
