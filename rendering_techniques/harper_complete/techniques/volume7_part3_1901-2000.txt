# CATEGORY VI: ENGINE INTEGRATION (1951-2000)

## 1951. Frame Graph
**Novelty Class:** Patent-Worthy

Declarative render pass system.

```cpp
void BuildFrameGraph(FrameGraph& fg) {
    // Declare resources
    auto depthBuffer = fg.Create<DepthBuffer>("Depth", g_Resolution, FORMAT_D32);
    auto gbufferA = fg.Create<RenderTarget>("GBufferA", g_Resolution, FORMAT_RGBA8);
    auto gbufferB = fg.Create<RenderTarget>("GBufferB", g_Resolution, FORMAT_RGBA16F);
    auto sceneColor = fg.Create<RenderTarget>("SceneColor", g_Resolution, FORMAT_RGBA16F);
    
    // Depth prepass
    fg.AddPass("DepthPrepass", [&](PassBuilder& pb) {
        pb.Write(depthBuffer);
        pb.Execute([](CommandList& cmd) {
            RenderDepthPrepass(cmd);
        });
    });
    
    // G-Buffer pass
    fg.AddPass("GBuffer", [&](PassBuilder& pb) {
        pb.Read(depthBuffer);
        pb.Write(gbufferA);
        pb.Write(gbufferB);
        pb.Execute([](CommandList& cmd) {
            RenderGBuffer(cmd);
        });
    });
    
    // Lighting pass
    fg.AddPass("Lighting", [&](PassBuilder& pb) {
        pb.Read(depthBuffer);
        pb.Read(gbufferA);
        pb.Read(gbufferB);
        pb.Write(sceneColor);
        pb.Execute([](CommandList& cmd) {
            RenderLighting(cmd);
        });
    });
    
    // Automatic barrier insertion, resource aliasing, async compute
    fg.Compile();
}
```

// Root signature with bindless heap access
#define BINDLESS_SRV_SPACE space0

Texture2D GetTexture(uint index) {
    return ResourceDescriptorHeap[index];

    return SamplerDescriptorHeap[index];
}

// Material data structure
struct MaterialData {
    uint albedoTexture;
    uint normalTexture;
    uint roughnessTexture;
    uint samplerIndex;
    float4 baseColor;
    float roughness;
    float metallic;
};

float4 SampleMaterial(MaterialData mat, float2 uv) {
    Texture2D albedo = GetTexture(mat.albedoTexture);
    SamplerState s = GetSampler(mat.samplerIndex);
    return albedo.Sample(s, uv) * mat.baseColor;
}

## 1953. Multi-Threaded Rendering
**Novelty Class:** Significant Improvement

Parallel command list recording.

```cpp
void RenderFrame() {
    // Per-thread command allocators
    std::vector<std::future<CommandList*>> futures;
    
    // Parallel recording
    for (uint i = 0; i < g_ThreadCount; i++) {
        futures.push_back(std::async([i]() {
            CommandList* cmd = g_CommandAllocators[i]->AllocateList();
            
            uint startObj = i * objectsPerThread;
            uint endObj = min(startObj + objectsPerThread, g_ObjectCount);
            
            for (uint obj = startObj; obj < endObj; obj++) {
                RecordObjectDraw(cmd, g_Objects[obj]);
            }
            
            cmd->Close();
            return cmd;
        }));
    }
    
    // Collect and execute
    std::vector<CommandList*> lists;
    for (auto& f : futures) {
        lists.push_back(f.get());
    }
    
    g_Queue->ExecuteCommandLists(lists.size(), lists.data());
}
```
