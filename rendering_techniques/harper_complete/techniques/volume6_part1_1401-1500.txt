## 1401. Thin G-Buffer Encoding
**Novelty Class:** Patent-Worthy

Pack full PBR material into 64 bits per pixel.

```hlsl
uint2 EncodeThinGBuffer(float3 normal, float roughness, float metallic, float3 albedo) {
    uint2 packed;
    // 32 bits: Octahedral normal (16) + roughness (8) + metallic (8)
    uint2 octNorm = PackOctahedral(normal);
    packed.x = (octNorm.x << 24) | (octNorm.y << 16) | 
               (uint(roughness * 255) << 8) | uint(metallic * 255);
    // 32 bits: RGB albedo (10-10-10) + 2 spare
    packed.y = (uint(albedo.r * 1023) << 22) | 
               (uint(albedo.g * 1023) << 12) | 
               (uint(albedo.b * 1023) << 2);
    return packed;
}
```

**Result:** 8 bytes/pixel vs 32+ for traditional G-buffer.

## 1405. Clustered Shading Optimizations
**Novelty Class:** Significant Improvement

AABB tightening and early-out patterns.

```hlsl
void BuildTightClusterAABB(uint3 clusterIdx, out float3 minPt, out float3 maxPt) {
    // Compute view-space AABB from cluster bounds
    float2 tileBoundsNDC = GetTileBoundsNDC(clusterIdx.xy);
    float nearZ = ClusterDepthSliceToViewZ(clusterIdx.z);
    float farZ = ClusterDepthSliceToViewZ(clusterIdx.z + 1);
    
    // Expand frustum corners to view space
    float3 corners[8];
    for (int i = 0; i < 4; i++) {
        float2 ndc = float2(tileBoundsNDC[i & 1], tileBoundsNDC[(i >> 1) & 1]);
        corners[i] = NDCToView(ndc, nearZ);
        corners[i + 4] = NDCToView(ndc, farZ);
    }
    
    // Tight AABB
    minPt = corners[0];
    maxPt = corners[0];
    for (int i = 1; i < 8; i++) {
        minPt = min(minPt, corners[i]);
        maxPt = max(maxPt, corners[i]);
    }
}

bool LightIntersectsCluster(Light light, float3 clusterMin, float3 clusterMax) {
    // Sphere-AABB test
    float3 closest = clamp(light.position, clusterMin, clusterMax);
    float distSq = dot(light.position - closest, light.position - closest);
    return distSq < light.radius * light.radius;
}
```

## 1451. Exponential Shadow Maps
**Novelty Class:** Significant Improvement

Single-sample soft shadows via exponential depth.

```hlsl
// Shadow map generation
float PS_ESMDepth(float4 pos : SV_Position) : SV_Target {
    return exp(g_ESMConstant * pos.z);
}

// Shadow evaluation
float EvaluateESM(float3 worldPos, float4x4 lightViewProj) {
    float4 lightSpace = mul(lightViewProj, float4(worldPos, 1));
    float2 shadowUV = lightSpace.xy / lightSpace.w * 0.5 + 0.5;
    float receiverDepth = lightSpace.z / lightSpace.w;
    
    float occluderExp = g_ESMMap.Sample(sampler, shadowUV);
    float receiverExp = exp(-g_ESMConstant * receiverDepth);
    
    return saturate(occluderExp * receiverExp);
}
```

**Result:** Soft shadows with single texture tap.

## 1452. Variance Shadow Maps
**Novelty Class:** Significant Improvement

Chebyshev's inequality for shadow filtering.

```hlsl
float2 PS_VSMDepth(float4 pos : SV_Position) : SV_Target {
    float depth = pos.z;
    return float2(depth, depth * depth);
}

float EvaluateVSM(float2 moments, float receiverDepth) {
    float variance = moments.y - moments.x * moments.x;
    variance = max(variance, g_MinVariance);
    
    float d = receiverDepth - moments.x;
    float pMax = variance / (variance + d * d);
    
    return (receiverDepth <= moments.x) ? 1.0 : pMax;
}
```

## 1455. Contact-Hardening Shadows
**Novelty Class:** Significant Improvement

Penumbra width from blocker distance.

```hlsl
float PCSS(float3 worldPos, float2 shadowUV, float receiverDepth) {
    // Step 1: Blocker search
    float avgBlockerDepth = 0;
    uint blockerCount = 0;
    
    for (int i = 0; i < 16; i++) {
        float2 offset = g_PoissonDisk[i] * g_SearchRadius;
        float sampleDepth = g_ShadowMap.Sample(sampler, shadowUV + offset);
        
        if (sampleDepth < receiverDepth) {
            avgBlockerDepth += sampleDepth;
            blockerCount++;
        }
    }
    
    if (blockerCount == 0) return 1.0;
    avgBlockerDepth /= blockerCount;
    
    // Step 2: Penumbra estimation
    float penumbraWidth = (receiverDepth - avgBlockerDepth) / avgBlockerDepth;
    penumbraWidth *= g_LightSize;
    
    // Step 3: PCF with variable kernel
    float shadow = 0;
    for (int i = 0; i < 32; i++) {
        float2 offset = g_PoissonDisk[i] * penumbraWidth;
        shadow += g_ShadowMap.SampleCmpLevelZero(cmpSampler, shadowUV + offset, receiverDepth);
    }
    
    return shadow / 32.0;
}
```

## 1503. Screen-Space Diffuse GI
**Novelty Class:** Significant Improvement

SSGI for one-bounce diffuse.

```hlsl
float3 ComputeSSGI(float2 uv, float3 worldPos, float3 normal) {
    float3 gi = 0;
    
    for (int i = 0; i < 16; i++) {
        // Cosine-weighted hemisphere sample
        float3 sampleDir = CosineSampleHemisphere(normal, g_BlueNoise[i]);
        
        // Screen-space ray march
        float3 hitPos;
        float2 hitUV;
        if (ScreenSpaceRayMarch(worldPos, sampleDir, hitPos, hitUV)) {
            float3 hitColor = g_SceneColor.Sample(sampler, hitUV).rgb;
            float3 hitNormal = g_GBufferNormal.Sample(sampler, hitUV).xyz;
            
            // Validate hit faces toward us
            if (dot(hitNormal, -sampleDir) > 0) {
                gi += hitColor;
            }
        }
    }
    
    return gi / 16.0;
}
```

## 1504. Radiosity Normal Mapping

float3 SampleRadiosityNormalMap(float2 lightmapUV, float3 tangentNormal) {
    // Sample 3 directional lightmaps (HL2 basis)
    float3 lightmap0 = g_Lightmap0.Sample(sampler, lightmapUV).rgb;
    float3 lightmap1 = g_Lightmap1.Sample(sampler, lightmapUV).rgb;
    float3 lightmap2 = g_Lightmap2.Sample(sampler, lightmapUV).rgb;
    
    // Basis directions in tangent space
    static const float3 basis0 = float3(-1/sqrt(6), 1/sqrt(2), 1/sqrt(3));
    static const float3 basis1 = float3(-1/sqrt(6), -1/sqrt(2), 1/sqrt(3));
    static const float3 basis2 = float3(sqrt(2/3), 0, 1/sqrt(3));
    
    // Reconstruct lighting from directional samples
    float3 lighting = lightmap0 * saturate(dot(tangentNormal, basis0)) +
                      lightmap1 * saturate(dot(tangentNormal, basis1)) +
                      lightmap2 * saturate(dot(tangentNormal, basis2));
    
    return lighting;
}
