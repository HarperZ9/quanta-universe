## 1551. Temporal Anti-Aliasing Core
**Novelty Class:** Significant Improvement

Production TAA with variance clipping.

```hlsl
float4 TAA(float2 uv) {
    // Current frame with jitter
    float4 current = g_CurrentColor.Sample(sampler, uv);
    
    // Motion vector lookup
    float2 velocity = g_MotionVectors.Sample(sampler, uv);
    float2 historyUV = uv - velocity;
    
    // History sample
    float4 history = g_HistoryColor.Sample(sampler, historyUV);
    
    // Neighborhood clamping (variance clip)
    float4 m1 = 0, m2 = 0;
    [unroll] for (int y = -1; y <= 1; y++) {
        [unroll] for (int x = -1; x <= 1; x++) {
            float4 s = g_CurrentColor.Sample(sampler, uv + float2(x, y) / g_Resolution);
            m1 += s;
            m2 += s * s;
        }
    }
    m1 /= 9.0;
    m2 /= 9.0;
    float4 sigma = sqrt(max(0, m2 - m1 * m1));
    
    float4 minC = m1 - sigma * g_ClipStrength;
    float4 maxC = m1 + sigma * g_ClipStrength;
    history = clamp(history, minC, maxC);
    
    // Blend
    float blend = 0.05; // 5% current, 95% history
    return lerp(history, current, blend);
}
```

```hlsl
float4 FXAA(float2 uv) {
    float4 center = g_Color.Sample(sampler, uv);
    float lumCenter = Luminance(center.rgb);
    
    float lumN = Luminance(g_Color.Sample(sampler, uv + float2(0, -1) / g_Resolution).rgb);
    float lumS = Luminance(g_Color.Sample(sampler, uv + float2(0, 1) / g_Resolution).rgb);
    float lumE = Luminance(g_Color.Sample(sampler, uv + float2(1, 0) / g_Resolution).rgb);
    float lumW = Luminance(g_Color.Sample(sampler, uv + float2(-1, 0) / g_Resolution).rgb);
    
    float lumMin = min(lumCenter, min(min(lumN, lumS), min(lumE, lumW)));
    float lumMax = max(lumCenter, max(max(lumN, lumS), max(lumE, lumW)));
    float lumRange = lumMax - lumMin;
    
    if (lumRange < max(g_EdgeThresholdMin, lumMax * g_EdgeThreshold))
        return center;
    
    // Edge direction
    float lumNS = lumN + lumS;
    float lumEW = lumE + lumW;
    bool isHorizontal = abs(lumNS - 2 * lumCenter) >= abs(lumEW - 2 * lumCenter);
    
    // Subpixel aliasing
    float subpixelOffset = abs((lumNS + lumEW) / 4 - lumCenter) / lumRange;
    subpixelOffset = saturate(subpixelOffset * g_SubpixelQuality);
    
    // Blend along edge
    float2 offset = isHorizontal ? float2(0, subpixelOffset / g_Resolution.y)
                                 : float2(subpixelOffset / g_Resolution.x, 0);
    
    return g_Color.Sample(sampler, uv + offset);
}
```

## 1601. Virtual Texturing Core
**Novelty Class:** Patent-Worthy

GPU-driven page fault handling.

```hlsl
float4 SampleVirtualTexture(float2 uv, float mip) {
    // Compute page ID
    uint2 pageCoord = uint2(uv * g_VirtualSize) >> (uint)mip;
    uint mipLevel = (uint)mip;
    
    // Lookup page table
    uint4 pageEntry = g_PageTable.Load(uint3(pageCoord, mipLevel));
    
    if (pageEntry.w == 0) {
        // Page not resident - request it
        RequestPage(pageCoord, mipLevel);
        // Fallback to lower mip
        return SampleVirtualTexture(uv, mip + 1);
    }
    
    // Compute physical texture UV
    float2 withinPage = frac(uv * g_VirtualSize / (1 << mipLevel));
    float2 physicalUV = (pageEntry.xy + withinPage) / g_PhysicalSize;
    
    return g_PhysicalTexture.Sample(sampler, physicalUV);
}
```

```hlsl
float ComputeStreamingPriority(uint textureId, float2 screenUV, float mipBias) {
    // Distance from screen center (favor center)
    float2 centered = screenUV - 0.5;
    float distFromCenter = length(centered);
    float centerWeight = 1 - saturate(distFromCenter * 2);
    
    // Mip level (favor detailed mips)
    float mipWeight = 1 - saturate(mipBias / 8);
    
    // Frame recency
    uint framesSinceUsed = g_FrameIndex - g_LastUsedFrame[textureId];
    float recencyWeight = exp(-framesSinceUsed * 0.1);
    
    return centerWeight * mipWeight * recencyWeight;
}
```

## 1652. Dual Quaternion Skinning

```hlsl
float4x4 DualQuatToMatrix(float4 real, float4 dual) {
    float4x4 m;
    
    float len2 = dot(real, real);
    float w = real.w, x = real.x, y = real.y, z = real.z;
    float t0 = dual.w, t1 = dual.x, t2 = dual.y, t3 = dual.z;
    
    m[0][0] = w*w + x*x - y*y - z*z;
    m[0][1] = 2*x*y - 2*w*z;
    m[0][2] = 2*x*z + 2*w*y;
    m[0][3] = 0;
    
    m[1][0] = 2*x*y + 2*w*z;
    m[1][1] = w*w - x*x + y*y - z*z;
    m[1][2] = 2*y*z - 2*w*x;
    m[1][3] = 0;
    
    m[2][0] = 2*x*z - 2*w*y;
    m[2][1] = 2*y*z + 2*w*x;
    m[2][2] = w*w - x*x - y*y + z*z;
    m[2][3] = 0;
    
    m[3][0] = 2*(-t0*x + t1*w - t2*z + t3*y);
    m[3][1] = 2*(-t0*y + t1*z + t2*w - t3*x);
    m[3][2] = 2*(-t0*z - t1*y + t2*x + t3*w);
    m[3][3] = len2;
```
